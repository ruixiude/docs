# 解析参数

/**
 * 路由参数接收示例
 *
 * 接收方式：
 * 1. 通过 @[Autowired] 注解，以及调用 [RXDRouter.inject] 方法进行自动注入
 * 2. 通过 [android.app.Activity.getIntent] 获取 Intent 对象，然后从 [android.content.Intent.getExtras] 获取参数集合
 */
@Route(path = ModuleRouteTable.RT_ROUTER_CASE_1)
class RouterCase1Activity : AppCompatActivity() {

    private lateinit var binding: ActivityRouterCase1Binding

    /*-------- 需要注意在 kotlin 下，@Autowired 标记的变量，访问范围不能为 private --------*/
    @Autowired // 自动将 key = "key" 的路由参数注入到 key 变量中
    @JvmField
    var key: String = ""

    @Autowired(name = "key") // 声明将 key = "key" 的路由参数注入到 otherName 变量中
    @JvmField
    var otherName: String = ""

    @Autowired
    @JvmField
    var test: Map<String, String> = emptyMap()

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityRouterCase1Binding.inflate(layoutInflater)
        setContentView(binding.root)

        RXDRouter.instance.inject(this) // 注入路由参数

        val extras = intent.extras
        if (extras != null) {
            val keySet = extras.keySet()
            val sb = StringBuilder()
            for (key in keySet) {
                sb.append(key).append("->").append(extras.get(key)).append(", ")
            }
            binding.tvParam.text = "Param: {$sb} \n testObj -> $test"
        } else {
            binding.tvParam.text = "Param is empty."
        }

    }

}
